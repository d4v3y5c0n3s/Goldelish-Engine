* root
** assets_core -> assets such as shaders & models so you can look at something
** demos -> examples using the engine
** include -> the engine's header files
*** SDL2 -> code for interfacing with SDL2
**** SDL_local.h -> see parent
*** assets -> code for dealing with assets such as skeletons and terrain
**** animation.h -> defines an animation as an array of frames & times
**** cmesh.h -> definition of 3D mesh in the game engine
**** config.h -> defines configuration files
**** effect.h -> defines a visual effect within the engine
**** font.h -> defines a font's data
**** image.h -> defines an image (and allows CPU-side image processing)
**** lang.h -> defines languages for the game to use
**** material.h -> definition of object materials
**** music.h -> defines music files
**** renderable.h -> defines a renderable mesh (may be rigged or not)
**** shader.h -> defines a GLSL shader program
**** skeleton.h -> defines a skeleton for 3D animations
**** sound.h -> defines a sound
**** terrain.h -> definition WIP heightmap-based terrain LOD system
**** texture.h -> defines textures
*** data -> code for datatypes such as dictionaries or splines
**** dict.h -> allows *void to be accessed with *char as keys
**** int_hashtable.h -> a hashtable for integers
**** int_list.h -> a dynamically resizable array of integers
**** list.h -> a dynamically expandable array of pointers
**** randf.h -> allows the creation of random floats
**** spline.h -> allows the creation of splines (think 3D curved lines)
**** vertex_hashtable.h -> hashtable for vertices, allows duplicate vertices to be checked for
**** vertex_list.h -> a dynamically resizable array of vertices
*** entities -> code for predefined entities such as light and particles
**** animated_object.h -> an object animated by a skeleton
**** camera.h -> a basic camera
**** instance_object.h -> a collection of static objects; supports instanced rendering
**** landscape.h -> object for rendering instance of terrain
**** light.h -> light object for lighting up scenes
**** particles.h -> displays particles
**** physics_object.h -> an object which utilizes the physics system
**** static_object.h -> an unmovable object that interacts with physics-affected objects
*** rendering -> code for the renderer
**** renderer.h -> a relatively simple deferred renderer implementation
**** sky.h -> allows the creation of a dynamic sky
*** ui -> code for the user interface system
**** ui_browser.h -> (unclear whatis) maybe a ui window of some kind?
**** ui_button.h -> button which may be clicked on
**** ui_dialog.h -> a popup window
**** ui_listbox.h -> used to display arrays of objects
**** ui_option.h -> same as button, but created in groups where only one may be selected
**** ui_rectangle.h -> a rectangle shape for ui
**** ui_slider.h -> used to slide through a range of values
**** ui_spinner.h -> allows a spinning icon to appear to inform the user that the game is loading
**** ui_style.h -> allows the appearance of the ui to be somewhat customized
**** ui_text.h -> allows the usage of text in the ui
**** ui_textbox.h -> a box which allows text to be entered
**** ui_toast.h -> a notifacation window
*** casset.h -> defines what an asset is in the context of the engine
*** caudio.h -> a basic audio playing system
*** cengine.h -> defines common functions and structs that are used throughout the engine
*** centity.h -> an interface for interacting with entities
*** cgraphics.h -> a basic layer for the graphics system
*** cjoystick.h -> a basic layer for SDL controls
*** cnet.h -> a basic networking layer
*** corange.h -> brings the different pieces of the engine together
*** cphysics.h -> a basic physics system definition
*** cui.h -> user interface manager

#  ATS compilers
PATSCC=$(PATSHOMEQ)/bin/patscc
PATSOPT=$(PATSHOMEQ)/bin/patsopt
PATSLIB=$(PATSHOMEQ)/ccomp/atslib/lib/libatslib.a

PLATFORM = $(shell uname)
ifeq ($(findstring Linux, $(PLATFORM)), Linux)
	DYNAMIC = libgoldelish.so
	STATIC = libgoldelish.a
	CFLAGS0 += -fPIC
	LDFLAGS0 += -lGL
endif
ifeq ($(findstring Darwin, $(PLATFORM)), Darwin)
	DYNAMIC = libgoldelish.so
	STATIC = libgoldelish.a
	CFLAGS0 +=  -fPIC
	LDFLAGS0 += -framework OpenGL
endif
ifeq ($(findstring MINGW, $(PLATFORM)), MINGW)
	DYNAMIC = goldelish.dll
	STATIC = libgoldelish.a
	LDFLAGS0 += -lmingw32 -lopengl32 -lSDL2main -lSDL2 -lSDL2_mixer -lSDL2_net -shared -g
endif

# - simply create rules for compiling ats to c, then .c to .o -
# - when compiling the c, remember to use 'PATSLIB' so that it can compile -
all: ats_compile $(DYNAMIC) $(STATIC)
$(DYNAMIC): $(OBJ)
	$(CC) $(OBJ) $(PATSLIB) $(LDFLAGS0) -o $@
$(STATIC): $(OBJ)
	$(AR) rcs $@ $(OBJ)
obj/%.o: src/%.dats.c | obj
	$(CC) $< -c $(CFLAGS0) -o $@
obj/%.o: src/*/%.dats.c | obj
	$(CC) $< -c $(CFLAGS0) -o $@
obj:
	mkdir obj

ats_src/*/%.dats.c: ats_src/*/%.dats
	$(PATSOPT) $(CFLAGS0) -o $@ $< $(LDFLAGS0)
